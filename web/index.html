Enable control buttons

import {Carousel} from 'caroucssel';

const el = document.querySelector('.carousel');
const carousel = new Carousel(el, {
	hasButtons: true
});

Enable pagination

import {Carousel} from 'caroucssel';

const el = document.querySelector('.carousel');
const carousel = new Carousel(el, {
	hasPagination: true
});

Use snap points

SASS

@import '~caroucssel/src/caroucssel';

.selector { // Use sass extend
	@extend %caroucssel;
}

.selector { // Use sass mixin
	@include caroucssel();
}

CSS

<link rel="stylesheet" href=href="./caroucssel.min.css">

<div class="caroucssel use-use-snap">
	<div>Item 1</div>
	<div>Item 2</div>
</div>
